if (navigator.userAgent.indexOf("Android") > -1) {
window.addEventListener("resize", (function () {
var dom, timer, initWindowHeight = window.innerHeight;
return function () {
var activeDom = document.activeElement;
if (activeDom && (activeDom.tagName.toString().toUpperCase() == "INPUT" || activeDom.tagName.toString().toUpperCase() == "TEXTAREA")) {
cancelAnimationFrame(timer);
timer = requestAnimationFrame(function () {
if (dom == null && window.innerHeight < initWindowHeight) {
dom = document.createElement("div");
dom.style.height = initWindowHeight - window.innerHeight + "px";
var contentDom = document.querySelector("#content");
if (contentDom) contentDom.appendChild(dom);
}
if (dom) {
//键盘弹起
if (window.innerHeight < initWindowHeight) {
dom.style.display = "block";
activeDom.scrollIntoViewIfNeeded(true);
} else dom.style.display = "none";
}
});
}
}
})(), false);
}
